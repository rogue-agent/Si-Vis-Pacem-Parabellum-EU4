# French Inheritance of Brittany
country_event = {
	id = french_inheritance.1
	title = "breton_inheritance.t"
	desc = "breton_inheritance.d"
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = FRA
		government = monarchy
		NOT = { war_with = BRI }
		BRI = {
			exists = yes
			is_subject = no
			is_neighbor_of = ROOT
			government = monarchy
		}
		is_subject = no
		num_of_cities = BRI
		OR = {
			AND = {
				FRA = { is_female = yes }
				BRI = { is_female = no }
			}
			AND = {
				FRA = { is_female = no }
				BRI = { is_female = yes }
			}
			BRI = {
				has_regency = yes
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}
	
	
	option = {		# France gains Restoration of Union casus belli on Brittany
		name = "breton_inheritance.a"
		ai_chance = { factor = 90 }
		FRA = {
			add_casus_belli = {	
				target = BRI
				type = cb_restore_personal_union
				months = 300
			}
		}
		BRI = {
			add_opinion = {
			who = FRA
			modifier = forced_breton_inheritance_brittany
			}
			hidden_effect = {
				country_event = { id = french_inheritance.4 days = 1 }
			}
		}
		every_known_country = {
			limit = {
				marriage_with = BRI
			}
			add_opinion = {
				who = ROOT
				modifier = forced_breton_inheritance_rm
			}
		}
	}
	option = {		# France rejects a personal union over Brittany
		name = "breton_inheritance.b"
		ai_chance = { factor = 10 }
		BRI = {
			add_opinion = {
				who = ROOT
				modifier = no_forced_breton_inheritance
			}
			hidden_effect = {
				country_event = { id = french_inheritance.5 days = 1 }
			}
		}
		add_prestige = -5
	}
}
