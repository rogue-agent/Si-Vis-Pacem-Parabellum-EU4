# Treaty of Gaeta (Aragonese Consideration)
country_event = {
    id = albania.1
    title = albania.1.t
    desc = albania.1.d
    picture = DIPLOMACY_eventPicture

    fire_only_once = yes

    trigger = {
        OR = {
            tag = ARA
            tag = NAP
        }
        is_subject = no
        if = {
            limit = {
                tag = ARA
            }
            OR = {
                senior_union_with = NAP
                121 = {
                    owned_by = ARA
                }
            }
        }
        ALB = {
            exists = yes
            is_subject = no
        }
        NOT = { is_year = 1500 }
        OR = {
            4174 = {
                owned_by = TUR
            }
            4175 = {
                owned_by = TUR
            }
            143 = {
                owned_by = TUR
            }
        }
    }

    mean_time_to_happen = {
		months = 60
	}

    option = {
        name = albania.1.a
        ai_chance = {
            factor = 80
            modifier = {
                factor = 100
                OR = {
                    is_rival = TUR
                    historical_rival_with = TUR
                }
            }
        }
        ALB = {
            country_event = {
                id = albania.2
                days = 30
            }
        }
    }

    option = {
        name = albania.1.b
        ai_chance = {
            factor = 20
        }
        add_prestige = -5
    }
}

# Treaty of Gaeta (Second Albanian Consideration)
country_event = {
    id = albania.5
    title = albania.5.t
    desc = albania.5.d
    picture = DIPLOMACY_eventPicture

    fire_only_once = yes

    major = yes

    trigger = {
        tag = ALB
        is_subject = no
        NOT = {
            war_with = TUR
        }
        AND = {
            143 = {
                NOT = {
                    controlled_by = TUR
                    owned_by = TUR
                }
            }
            4174 = {
                NOT = {
                    controlled_by = TUR
                    owned_by = TUR
                }
            }
            4175 = {
                owned_by = ALB
                NOT = {
                    controlled_by = TUR
                }
            }
        }
        OR = {
            ARA = {
                has_country_flag = treaty_of_gaeta_successful
                is_subject = no
            }
            NAP = {
                has_country_flag = treaty_of_gaeta_successful
                is_subject = no
            }
        }
    }

    mean_time_to_happen = {
		days = 30
	}

    option = {
        name = albania.5.a
        ai_chance = {
            factor = 50
            modifier = {
                factor = 70
                alliance_with = ARA
                ARA = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
            modifier = {
                factor = 70
                alliance_with = NAP
                NAP = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
        }

        if = {
            limit = {
                ARA = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
            ARA = {
                create_march = ALB
                add_opinion = {
				    who = ALB
				    modifier = honored_treaty_of_gaeta
			    }
                country_event = {
                    id = albania.6
                    days = 1
                }
            }
        }
        if = {
            limit = {
                NAP = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
            NAP = {
                create_march = ALB
                add_opinion = {
				    who = ALB
				    modifier = honored_treaty_of_gaeta
			    }
                country_event = {
                    id = albania.6
                    days = 1
                }
            }
        }
    }

    option = {
        name = albania.5.b
        ai_chance = {
            factor = 50
            modifier = {
                factor = 30
                alliance_with = ARA
                ARA = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
            modifier = {
                factor = 30
                alliance_with = NAP
                NAP = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
        }

        remove_country_modifier = treaty_of_gaeta
        add_country_modifier = {
            name = broke_treaty_of_gaeta
            duration = 3650
        }
        if = {
            limit = {
                ARA = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
            ARA = {
                add_opinion = {
				    who = ALB
				    modifier = dishonored_treaty_of_gaeta
			    }
                remove_opinion = {
                    who = ALB
                    modifier = accepted_treaty_of_gaeta
                }
                country_event = {
                    id = albania.7
                    days = 1
                }	
            }
        }
        if = {
            limit = {
                NAP = {
                    has_country_flag = treaty_of_gaeta_successful
                }
            }
            NAP = {
                add_opinion = {
				    who = ALB
				    modifier = dishonored_treaty_of_gaeta
			    }
                remove_opinion = {
                    who = ALB
                    modifier = accepted_treaty_of_gaeta
                }
                country_event = {
                    id = albania.7
                    days = 1
                }	
            }
        }
    }
}
