# Death of the Albertine Line
country_event = {

	title = "death_albertine_line.t"
	desc = "death_albertine_line.d"
	picture = CITY_VIEW_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		tag = STY
		dynasty = "von Habsburg"
		AUT = {
			exists = yes
			ai = yes
		}
		HUN = {
			NOT = {
				has_heir = "Ladislaus Postumus"
				has_ruler = "Ladislaus Postumus"
				has_country_flag = interregnum_flag
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

    immediate = {
        set_country_flag = inherited_austria_1
		set_country_flag = inherited_austria_2

		hidden_effect = {
			if = {
				limit = {
					HUN = {
						is_lesser_in_union = yes
						is_subject_of = AUT
					}
				}
				HUN = {
					grant_independence = yes
				}
				CRO = {
					set_country_flag = croatia_inherited
				}
			}
			if = {
				limit = {
					AUT = {
						has_casus_belli = {
							type = cb_restore_personal_union
							target = HUN
						}
					}
				}
				CRO = {
					set_country_flag = croatia_inherited
				}
			}
			if = {
				limit = {
					BOH = {
						is_lesser_in_union = yes
						is_subject_of = AUT
					}
				}
				BOH = {
					grant_independence = yes
				}
			}
		}
    }
	
	option = {		#Styria inherits Austria
		name = "death_albertine_line.a"
		hidden_effect = {
			if = {
				limit = {
					CRO = {
						has_country_flag = croatia_inherited
						OR = {
							is_subject_of = AUT
							AUT = {
								has_casus_belli = {
									type = cb_restore_personal_union
									target = CRO
								}
							}
						}
					}
				}
				CRO = {
					grant_independence = yes
				}
			}
			AUT = {
				every_subject_country = {
					limit = {
						is_lesser_in_union = yes
						NOT = {
							tag = CRO
						}
					}
					country_event = {
						id = austria_hungary.33
						days = 1
					}
				}
			}
		}
		AUT = {
			clr_country_flag = ladislaus_returns
		}
		inherit = AUT
		hidden_effect = {
			HUN = {
            	country_event = {
                	id = austria_hungary.7
                	days = 30
            	}
        	}
		}
	}
}

					}

			}

		custom_tooltip = austria_hungary.9.a.tooltip

			add_opinion = {
				who = ROOT

			}	
		}

				region = low_countries_region
			}
