#Hussites march on Prague
country_event = {
	id = flavor_boh.1
	title = flavor_boh.1.t
	desc = flavor_boh.1.d
	picture = CIVIL_WAR_eventPicture

	major = yes

	fire_only_once = yes

	trigger = {
		tag = BOH
		owns = 266
		is_year = 1446
		NOT = { is_year = 1457 }
		is_at_war = no
		religion = catholic
		is_free_or_tributary_trigger = yes
	}
	
	mean_time_to_happen = {
		months = 17
	}
	
	option = {
		ai_chance = { factor = 90 }
		name = flavor_boh.1.a
		add_papal_influence = -100
		if = {
			limit = {
				has_regency = no
			}
			tooltip = {
				kill_ruler = yes
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = yes
		}
		define_ruler = {
			name = "Jirí"
			dynasty = "z Podebrad"
			adm = 5
			dip = 3
			mil = 3
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = zealot_personality }
		}
		add_reform_desire = 0.03
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		set_country_flag = boh_supported_hussites
		set_country_flag = boh_hussites_mission_accept
		PAP = {
			country_event = { id = flavor_boh.9 }
		}
		every_known_country = {
			limit = {
				religion = catholic
				is_free_or_tributary_trigger = yes
				OR = {
					is_neighbor_of = ROOT
					num_of_cities = 15
					government = theocracy
				}
			}
			add_opinion = {
				who = root
				modifier = boh_heretic_hussite
			}
		}
	}
	
	option = {
		name = flavor_boh.1.b
		if = {
			limit = {
				has_regency = no
			}
			tooltip = {
				kill_ruler = yes
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = yes
		}
		define_ruler = {
			name = "Oldrich II"
			dynasty = "z Rozmberka"
			max_random_adm = 3
			max_random_dip = 3
			max_random_mil = 3
		}
		define_heir = {
			name = "Ladislaus Postumus"
			dynasty = "von Habsburg"
			culture = austrian
			adm = 3
			dip = 3
			mil = 2
			birth_date = 1440.2.22
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = well_connected_personality }
		}
		set_country_flag = boh_hussites_mission_decline
		if = {
			limit = { exists = PAP }
			add_papal_influence = 5
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
		266 = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Jirí z Podebrad"
			}		
		}
		random_owned_province = {
			limit = {
				NOT = { province_id = 266 }
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 1
			}
		}
	}
}
