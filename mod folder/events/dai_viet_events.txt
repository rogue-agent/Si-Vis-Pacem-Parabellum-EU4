# Lê Bang Co Dies Early
country_event = {
	id = dai_viet.23
	title = dai_viet.23.t
	desc = dai_viet.23.d
	picture = DEATH_OF_HEIR_eventPicture

	fire_only_once = yes

	trigger = {
		tag = DAI
		has_ruler = "Thi Anh"
		dynasty = "Nguyên"
		has_ruler_flag = empress_dowager
		NOT = {
			has_heir = "Bang Co"
		}
		NOT = {
			is_year = 1457
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = dai_viet.23.a
		ai_chance = {
			factor = 75
		}
		define_heir = {
			name = "Nghi Dân"
			dynasty = "Lê"
			max_random_adm = 0
			max_random_dip = 0
			max_random_mil = 0
			birth_date = 1439.6.1
		}
		kill_ruler = yes
		if = {
			limit = {
				has_ruler = "Nghi Dân"
			}
			hidden_effect = {
				add_ruler_personality = cruel_personality
			}
		}
		set_country_flag = nghi_dan_cruelty
	}

	option = {
		name = dai_viet.23.b
		ai_chance = {
			factor = 25
		}
		add_stability = -3
		capital_scope = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Nghi Dân Lê"
			}
		}
	}
}
