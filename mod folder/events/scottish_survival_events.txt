# The Rough Wooing
country_event = {
	id = scottish_survival.3
	title = scottish_survival.3.t
	desc = scottish_survival.3.d
	picture = ACCUSATION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = ENG
		government = monarchy
		NOT = { war_with = SCO }
		SCO = {
			exists = yes
			is_subject = no
			is_neighbor_of = ROOT
			government = monarchy
		}
		is_subject = no
		num_of_cities = SCO
		OR = {
			AND = {
				ENG = { is_female = yes }
				SCO = { is_female = no }
			}
			AND = {
				ENG = { is_female = no }
				SCO = { is_female = yes }
			}
			SCO = {
				has_regency = yes
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}
	
	
	option = {		# England gains Restoration of Union casus belli on Scotland
		name = "scottish_survival.3.a"
		ai_chance = { factor = 90 }
		ENG = {
			add_casus_belli = {	
				target = SCO
				type = cb_restore_personal_union
				months = 300
			}
		}
		SCO = {
			add_opinion = {
				who = ENG
				modifier = laid_claim_to_scotland
			}
			hidden_effect = {
				country_event = { id = scottish_survival.4 days = 1 }
			}
		}
		every_known_country = {
			limit = {
				marriage_with = SCO
			}
			add_opinion = {
				who = ROOT
				modifier = laid_claim_to_scotland_rm
			}
		}
	}
	option = {		# England rejects a personal union over Scotland
		name = "scottish_survival.3.b"
		ai_chance = { factor = 10 }
		add_prestige = -5
	}
}
