# Prussian Confederation rises up
country_event = {
	id = flavor_teu.1
	title = "flavor_teu.EVTNAME1"
	desc = "flavor_teu.EVTDESC1"
	picture = BATTLE_VISTULA_LAGOON_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1450
		NOT = { is_year = 1500 }
		tag = TEU
		NOT = { has_country_flag = pru_confederation_curtailed }
		POL = {
			NOT = {
				truce_with = TEU
			}
		}
		has_country_flag = pru_confederation_flag
		is_subject = no
		exists = POL
		is_neighbor_of = POL
		is_at_war = no
		NOT = { alliance_with = POL }
		POL = { is_subject = no }
		POL = { num_of_cities = 10 }
		num_of_cities = 5
		OR = {
			NOT = { exists = DNZ }
			DNZ = {
				is_subject_of = ROOT
			}
		}
		OR = {
			AND = {
				owns = 43
				controls = 43
			}
			DNZ = { owns = 43 controls = 43 }
		}
		estate_influence = {
			estate = estate_burghers
			influence = 50
		}
		any_owned_province = {
			has_estate = estate_burghers
		}
	}
	
	mean_time_to_happen = {
		months = 60
		modifier = {
			factor = 0.5
			is_year = 1454
		}
		modifier = {
			factor = 0.1
			is_year = 1466
		}
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = pru_confederation_happened
			every_owned_province = {
				limit = {
					has_estate = estate_burghers
					controlled_by = ROOT
				}
				add_core = DNZ
			}
			if = {
				limit = {
					NOT = {
						exists = DNZ
					}
				}
				release = DNZ
				DNZ = {
					change_government = republic
					add_government_reform = oligarchy_reform
					adopt_reform_progress = ROOT
				}
				DNZ = {
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.6
							cavalry = 0.3
						}
					}
					random_owned_province = {
						limit = {
							has_port = yes
						}
						build_to_forcelimit = {
							galley = 0.3
							light_ship = 0.3
							transport = 0.2
						}
					}
				}
			}
			43 = {
				add_core = DNZ
			}
			2997 = {
				add_core = DNZ
			}
			1841 = {
				add_core = DNZ
			}
			1859 = {
				add_core = DNZ
			}
			42 = {
				add_core = DNZ
			}
			POL = {
				vassalize = DNZ
			}
		}
	}
	
	option = { # Go to war
		name = "flavor_teu.1.a"
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				OR = {
					is_bankrupt = yes
					war_exhaustion = 10
				}
			}
		}
		custom_tooltip = flavor_teu.1.tooltip
		hidden_effect = {
			POL = {
				country_event = { id = flavor_teu.2 }
			}
		}
	}
	option = { # Let them go
		name = "flavor_teu.1.b"
		ai_chance = { 
			factor = 100 
			modifier = {
				factor = 0
				OR = {
					is_bankrupt = no
					NOT = { war_exhaustion = 10 }
				}
			}
		}
		add_stability = -1
		add_prestige = -20
	}
}
